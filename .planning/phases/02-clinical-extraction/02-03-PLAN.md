       ```python
       import pytest
       from dental_notes.clinical.models import ExtractionResult, SpeakerChunk

       pytestmark = pytest.mark.integration

       class TestOllamaConnectivity:
           def test_ollama_available(self, ollama_service):
               assert ollama_service.is_available()

           def test_model_ready(self, ollama_service):
               assert ollama_service.is_model_ready()

       class TestClinicalExtraction:
           def test_extraction_produces_soap_note(self, clinical_extractor, sample_transcript):
               result = clinical_extractor.extract(sample_transcript)
               assert isinstance(result, ExtractionResult)
               assert result.soap_note.subjective
               assert result.soap_note.objective
               assert result.soap_note.assessment
               assert result.soap_note.plan

           def test_extraction_cdt_codes_valid(self, clinical_extractor, sample_transcript):
               result = clinical_extractor.extract(sample_transcript)
               assert len(result.soap_note.cdt_codes) > 0
               for code in result.soap_note.cdt_codes:
                   assert code.code.startswith("D")
                   assert len(code.code) == 5

           def test_extraction_filters_chitchat(self, clinical_extractor, sample_transcript):
               result = clinical_extractor.extract(sample_transcript)
               soap_text = (result.soap_note.subjective + result.soap_note.objective +
                           result.soap_note.assessment + result.soap_note.plan).lower()
               assert "how are you" not in soap_text
               assert "doing well" not in soap_text

       class TestSpeakerReattribution:
           def test_reattribution_preserves_chunks(self, speaker_reattributor, sample_chunks):
               result = speaker_reattributor.reattribute(sample_chunks)
               assert len(result) == len(sample_chunks)

           def test_reattribution_labels_valid(self, speaker_reattributor, sample_chunks):
               result = speaker_reattributor.reattribute(sample_chunks)
               for chunk in result:
                   assert chunk.speaker in ("Doctor", "Patient")

       class TestFullPipeline:
           def test_extraction_then_reattribution(self, clinical_extractor, speaker_reattributor, sample_chunks, sample_transcript):
               extraction = clinical_extractor.extract(sample_transcript)
               assert isinstance(extraction, ExtractionResult)
               reattributed = speaker_reattributor.reattribute(sample_chunks)
               assert len(reattributed) == len(sample_chunks)

           def test_unload_after_extraction(self, ollama_service, clinical_extractor, sample_transcript):
               clinical_extractor.extract(sample_transcript)
               ollama_service.unload()  # Should not raise
       ```
